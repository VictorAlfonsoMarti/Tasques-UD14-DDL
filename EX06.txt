CREATE DATABASE EJERCICIO_6;
USE EJERCICIO_6;
GO
CREATE TABLE Peliculas (
	Codigo INT PRIMARY KEY IDENTITY(1,1),
	Nombre NVARCHAR(100), 
	CalificacionEdad INT
)
CREATE TABLE Salas (
	Codigo INT PRIMARY KEY IDENTITY(1,1),
	Nombre NVARCHAR(100),
	Pelicula INT FOREIGN KEY REFERENCES Peliculas(Codigo) ON DELETE SET NULL ON UPDATE CASCADE
)
