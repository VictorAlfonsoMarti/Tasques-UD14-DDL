CREATE DATABASE EJERCICIO_3;
USE EJERCICIO_3;
GO
CREATE TABLE Fabricantes (
	Codigo INT PRIMARY KEY IDENTITY(1,1),
	Nombre NVARCHAR(100)
)
CREATE TABLE Articulos (
	Codigo INT PRIMARY KEY IDENTITY(1,1),
	Nombre NVARCHAR(100),
	Precio INT,
	Fabricante INT FOREIGN KEY REFERENCES Fabricantes(Codigo)
)
