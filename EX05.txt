CREATE DATABASE EJERCICIO_5;
USE EJERCICIO_5;
GO
CREATE TABLE Almacenes (
	Codigo INT PRIMARY KEY IDENTITY(1,1),
	Lugar NVARCHAR(100),
	Capacidad INT
)
CREATE TABLE Cajas (
	NumReferencia CHAR(5) PRIMARY KEY NOT NULL,
	Contenido NVARCHAR(100),
	Valor INT,
	Almacen INT FOREIGN KEY REFERENCES Almacenes(Codigo) ON DELETE SET NULL ON UPDATE CASCADE 
)
