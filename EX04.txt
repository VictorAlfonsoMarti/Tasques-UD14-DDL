CREATE DATABASE EJERCICIO_4;
USE EJERCICIO_4;
GO
CREATE TABLE Departamentos (
	Codigo INT PRIMARY KEY IDENTITY (1,1),
	Nombre NVARCHAR(100),
	Presupuesto INT
)
CREATE TABLE Empleados (
	DNI VARCHAR(8) PRIMARY KEY NOT NULL, 
	Nombre NVARCHAR(100),
	Apellidos NVARCHAR(255),
	Departamento INT FOREIGN KEY REFERENCES Departamentos(Codigo) ON DELETE SET NULL ON UPDATE CASCADE
);
